from fpdf import FPDF class SEOReportPDF(FPDF): def header(self): self.set_font("Arial", "B", 14) self.cell(0, 10, "SEO InsightHub Report", ln=True, align="C") def body(self, tips): self.set_font("Arial", size=12) for line in tips.split('\n'): self.multi_cell(0, 10, line) def create_pdf(tips, filename="seo_report.pdf"): pdf = SEOReportPDF() pdf.add_page() pdf.body(tips) pdf.output(filename) return filename
